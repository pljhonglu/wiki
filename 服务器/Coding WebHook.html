wiki_hook.sh

#!/bin/bash
HOOKLOGPATH='wiki_hook.log'
GITLOGPATH='wiki_git.log'
SIMIKILOGPATH='wiki_simiki.log'

cd Wiki
date=`date "+%Y-%m-%d %H:%M:%S"`

echo "====================pull start ${date}===================" >> $HOOKLOGPATH
git pull > $GITLOGPATH
echo "git pull finish!" >> $HOOKLOGPATH

echo "${date}" > $SIMIKILOGPATH
simiki g >> $SIMIKILOGPATH
echo "simiki generate finish!" >> $HOOKLOGPATH

echo "to output Directory." >> $HOOKLOGPATH
cd Wiki/output
git add --update 
git commit -m "Update Documentation"
git push -u origin gh-pages

echo "simiki output file push finished!" >> $HOOKLOGPATH
echo "==================== done ===================" >> $HOOKLOGPATH
echo "finish"
